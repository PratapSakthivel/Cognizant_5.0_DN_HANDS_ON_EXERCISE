ENROLLMENT REDUCER (enrollment.reducer.ts):
--------------------------------------------
import { createReducer, on } from '@ngrx/store';
import * as EnrollmentActions from './enrollment.actions';

export const enrollmentReducer = createReducer(
  initialEnrollmentState,

  // Enroll in course
  on(EnrollmentActions.enrollInCourse, (state, { courseId }) => ({
    ...state,
    enrolledCourseIds: [...state.enrolledCourseIds, courseId]  // Immutable add
  })),

  // Unenroll from course
  on(EnrollmentActions.unenrollFromCourse, (state, { courseId }) => ({
    ...state,
    enrolledCourseIds: state.enrolledCourseIds.filter(id => id !== courseId)
  })),

  // Set all enrolled courses
  on(EnrollmentActions.setEnrolledCourses, (state, { courseIds }) => ({
    ...state,
    enrolledCourseIds: courseIds
  }))
);

COMPONENT USAGE (course-card.component.ts):
--------------------------------------------
export class CourseCardComponent {
  @Input() course!: Course;
  private store = inject(Store);

  // Select enrollment state
  enrolledIds$ = this.store.select(
    EnrollmentSelectors.selectEnrolledIds
  );

  // Check if this specific course is enrolled
  isEnrolled$ = this.store.select(
    EnrollmentSelectors.isEnrolledInCourse(this.course.id)
  );

  onEnroll() {
    this.store.dispatch(
      EnrollmentActions.enrollInCourse({ courseId: this.course.id })
    );
  }

  onUnenroll() {
    this.store.dispatch(
      EnrollmentActions.unenrollFromCourse({ courseId: this.course.id })
    );
  }
}

TEMPLATE (course-card.component.html):

  <!-- Show Enroll button if NOT enrolled -->
  <button
    *ngIf="!(isEnrolled$ | async)"
    (click)="onEnroll()"
    class="btn-primary">
    Enroll
  </button>

  <!-- Show Unenroll button if enrolled -->
  <button
    *ngIf="isEnrolled$ | async"
    (click)="onUnenroll()"
    class="btn-danger">
    ✓ Enrolled - Unenroll
  </button>
</div>

SELECTOR (enrollment.selectors.ts):
------------------------------------
// Select enrolled course IDs
export const selectEnrolledIds = createSelector(
  selectEnrollmentState,
  (state) => state.enrolledCourseIds
);

// Check if specific course is enrolled (parameterized selector)
export const isEnrolledInCourse = (courseId: number) => createSelector(
  selectEnrolledIds,
  (enrolledIds) => enrolledIds.includes(courseId)
);

// Cross-slice selector: get enrolled courses with full details
export const selectEnrolledCourses = createSelector(
  selectAllCourses,        // From course slice
  selectEnrolledIds,       // From enrollment slice
  (courses, enrolledIds) => 
    courses.filter(c => enrolledIds.includes(c.id))
);

STATE BENEFITS:
---------------
✅ Single source of truth for enrollment
✅ Enrollment state separate from course state
✅ Easy to check enrollment status
✅ Immutable updates (no array mutation)
✅ Time-travel debugging (can undo/redo enrollments)
✅ Testable selectors

ALTERNATIVE APPROACHES (WHY NGRX IS BETTER):
---------------------------------------------
❌ Service with BehaviorSubject:
   - Hard to debug (no action history)
   - No time-travel
   - Easy to mutate state accidentally

❌ Component state:
   - Not shared across components
   - Lost on navigation
   - No centralized state

✅ NgRx Store:
   - Centralized, immutable state
   - Full action history
   - Time-travel debugging
   - Predictable state updates
